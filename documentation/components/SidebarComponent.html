  ngOnInit(): void {
    this.cpEventsService.toggleSidebar
      .pipe(takeUntilDestroyed(this.destroyRef))
      .subscribe((res: boolean) => {
        this.menuOpen = res;
    })
  }

  changeLanguage(): void {
    this.utilityService.changeLanguage(this.currentLanguage, this.userData?.uuid);
  }

  toggleMenu(): void {
    this.menuOpen = !this.menuOpen;  
  }
